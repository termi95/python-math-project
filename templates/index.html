 val = (q.y - p.y) * (r.x - q.x) - (q.x - p.x) * (r.y - q.y)

def on_segment(p: Point, q: Point, r: Point):
 return (
   q.x <= max(p.x, r.x)
   and q.x >= min(p.x, r.x)
  and q.y <= max(p.y, r.y)
  and q.y >= min(p.y, r.y)
